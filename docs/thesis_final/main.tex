% Păstrăm noțiunea de capitol, dar o tipărim ca titlu continuu, fără clearpage.
\RenewDocumentCommand{\chapter}{s m}{%
  \par\addvspace{2.5ex plus .5ex minus .2ex}%
  \IfBooleanTF{#1}{%
    \noindent{\normalfont\Large\bfseries #2}\par\nobreak\vspace{1.5ex plus .2ex}%
  }{%
    \refstepcounter{chapter}%
    \setcounter{section}{0}%
    \setcounter{subsection}{0}%
    \addcontentsline{toc}{chapter}{\protect\numberline{\thechapter}#2}%
    \noindent{\normalfont\Large\bfseries Capitolul~\thechapter. #2}\par\nobreak\vspace{1.5ex plus .2ex}%
  }%
}
